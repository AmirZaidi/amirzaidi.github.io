Een OOP gebruik van Templates

template_class.php
<?php
class template
{
	private $content;
	private $vars = Array();

	function __construct($tplfile)
	{
		ob_start();
		require $tplfile . '.tpl';
		$this->content = ob_get_clean();
	}

	public function setVar($key, $value)
	{
		$this->vars['{' . $key . '}'] = $value;
	}

	public function output()
	{
		echo '<!DOCTYPE html>',
		'<head>',
			$this->getheader(),
		'</head>',
		'<body>',
			strtr($this->vars, $this->content),
		'</body>',
		'</html>';
	}

	private function getheader()
	{
		return '<title>Welkom op Kidi!</title>';
	}
}
?>				
index.php
<?php
$page = 'index';
if (!empty($_GET['page']))
{
	$page = $_GET['page'];
}

require 'template_class.php';

$tpl = new template($page);

$tpl->setVar('tijd', date('H:i:s'));
$tpl->setVar('naam', 'Roens');
$tpl->setVar('websitenaam', 'Kidi');

$tpl->output();
?>
