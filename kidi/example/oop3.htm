using System;
class Program
{
	static void VeranderTekst(string Tekst)
	{
		Tekst = "Veranderd";
	}

	static void VeranderTekstHouder(TekstHouder Houder)
	{
		Houder.Tekst = "Veranderd";
	}

	static void Main(string[] args)
	{
		string HalloTekst = "Hallo";

		TekstHouder HoiTekst = new TekstHouder();
		HoiTekst.Tekst = "Hoi";

		VeranderTekst(HalloTekst);
		VeranderTekstHouder(HoiTekst);

		Console.WriteLine(HalloTekst);
		Console.Write(HoiTekst.Tekst);
	}
}					
using System;
class TekstHouder
{
	public string Tekst;
}
