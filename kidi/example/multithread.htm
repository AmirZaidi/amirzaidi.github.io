Parallele Console.Write

using System;
class Program
{
	static void Main(string[] args)
	{
		Thread NewThread = new Thread(SchrijfB);
		NewThread.Start();

		for (int i = 0; i < 10; i++)
		{
			Console.Write("a");
		}
	}

	static void SchrijfB()
	{
		for (int i = 0; i < 10; i++)
		{
			Console.Write("b");
		}
	}
}
