Meerdere Gebruikers

using System;
class Gebruiker
{
	public string Voornaam;
	public string Achternaam;
	public int Leeftijd;

	public Gebruiker(string NieuweVoornaam, string NieuweAchternaam, int NieuweLeeftijd)
	{
		this.Voornaam = NieuweVoornaam;
		this.Achternaam = NieuweVoornaam;
		this.Leeftijd = NieuweLeeftijd;
	}

	public string VolledigeNaam()
	{
		return (this.Voornaam + " " + this.Achternaam);
	}
}				
using System;
class Program
{
	static void Main(string[] args)
	{
		Gebruiker[] Gebruikers = new Gebruiker[20];

		Gebruikers[0] = new Gebruiker("Rens", "Kievit", 3);
		Gebruikers[1] = new Gebruiker("Amir", "Zaidi", 3);

		foreach (Gebruiker Persoon in Gebruikers)
		{
			if (Persoon == null)
			{
				continue;
			}

			Console.WriteLine(Persoon.VolledigeNaam() + " is " + (string)Persoon.Leeftijd + " jaar oud.");
		}
	}
}

Omdat een aantal waardes in de array nog niet een echte waarde hebben gekregen zijn ze nog null. Null wilt zeggen dat die ruimte wel in je geheugen gereserveerd is en die variabel daarheen wijst, maar die ruimte nog geen echte waarde heeft gekregen.
