classes/site.php
<?php
class site
{
	public static function getRemoteIp()
	{
		//Hier komt een cloudflare script als je dat wilt gebruiken
		return $_SERVER['REMOTE_ADDR'];
	}

	public static function redirect($url)
	{
		header('Location: ' . $url);
		exit();
	}
}
?>				
classes/user.php
<?php
class user
{
	public $name, $id, $mail;

	function __construct($name, $data)
	{
		$this->name = $name;
		$this->id = $data['id'];
		$this->mail = $data['mail'];
	}
}
?>
